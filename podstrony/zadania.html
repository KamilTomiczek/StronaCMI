#include <iostream>
using namespace std;

                int main()
                {
                    setlocale(LC_ALL, "");
                    int n,k;
                    int p,m;
                    char znak;
                    p = 0;
                    n = 0;
                    k = 0;
                    m = 0;
                    cout << "Podaj ilość poleceń" << endl;
                    cin >> n;
                    int * tablica = new int[n];
                    system("cls");
                    cout << "Wpisz polecenie" << endl;
                    cin >> znak >> k;
                    system("cls");
                    if (znak == 'R') {
                        tablica[n - 1] == k;
                    }
                    if(znak == 'L')
                    {
                        tablica[n - n] == k;
                    }
                    for (int i = 1; i < n; i++)
                    {
                        cin >> znak >> k;
                        if (znak == 'R')
                        {
                            for (int j = n - 1; j >= 0; j++)
                            {
                                tablica[j - p] == tablica[j - p - 1];
                                p = p + 1;
                            }
                            tablica[n - 1] == k;
                        }
                        if(znak == 'L')
                        {
                            for (int j = 0; j < n; j++)
                            {
                                tablica[j + p] == tablica[j + p + 1];
                                p = p + 1;
                            }
                            tablica[0] == k;
                        }
                        if (znak == '?')
                        {
                            for (int i = 0; i < n; i++)
                            {
                                if (tablica[i] == k) {
                                    m = i;
                                    break;
                                }
                                else                {
                                    continue;
                                }
                            }
                        }
                        if (m = n / 2) {
                            int q = n / 2;
                            cout << "Wynik" << q << endl;
                        }
                        else        {
                            if (m > n / 2) {
                                int q = m - n;
                                cout << "Wynik" << q << endl;
                            }
                            else            {
                                if (m == 0) {
                                    cout << "0" << endl;
                                }
                                else                {
                                    int q = m - m;
                                    cout << "Wynik" << q << endl;
                                }
                            }
                        }
                    }
                }
